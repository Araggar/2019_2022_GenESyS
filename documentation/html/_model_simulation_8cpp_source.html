41 void ModelSimulation::startSimulation() {
42  if (!_model->checkModel()) {
43  _model->getTracer()->trace(Util::TraceLevel::errors, "Model check failed. Cannot start simulation.");
44  return;
45  }
46  _initSimulation();
47  _model->getOnEventManager()->NotifySimulationStartListeners(new SimulationEvent(0, nullptr));
48 
49 
50  for (_currentReplicationNumber = 1; _currentReplicationNumber <= _info->getNumberOfReplications(); _currentReplicationNumber++) {
51  _initReplication();
52  _model->getOnEventManager()->NotifyReplicationStartListeners(new SimulationEvent(_currentReplicationNumber, nullptr));
53 
54  while (!_finishReplicationCondition()) {
55  _stepSimulation();
56  if (_pauseOnEvent) {
57  std::cout << "[paused] ...press any key to continue...";
58  std::cin.get();
59  std::cout << std::endl;
60  }
61  }
62 
63  _model->getOnEventManager()->NotifyReplicationEndListeners(new SimulationEvent(_currentReplicationNumber, nullptr));
64 
65 
66  std::string causeTerminated = "";
67  if (_model->getEvents()->empty()) {
68  causeTerminated = "event queue is empty";
69  } else if (_stopRequested) {
70  causeTerminated = "user requested to stop";
71  } else if (!(this->_simulatedTime < _info->getReplicationLength())) {
72  causeTerminated = "replication length " + std::to_string(_info->getReplicationLength()) + " was achieved";
73  } else if (_model->parseExpression(_info->getTerminatingCondition())) {
74  causeTerminated = "termination condition was achieved";
75  } else causeTerminated = "unknown";
76  std::string message = "Replication " + std::to_string(_currentReplicationNumber) + " of " + std::to_string(_info->getNumberOfReplications()) + " has finished at time " + std::to_string(_simulatedTime) + " because " + causeTerminated;
77  _model->getTracer()->trace(Util::TraceLevel::simulation, message);
78  _showReplicationStatistics();
79  }
80  _showSimulationStatistics();
81 
82  std::cout << "Simulation has finished.\n";
83  _model->getOnEventManager()->NotifySimulationEndListeners(new SimulationEvent(0, nullptr));
84 
85 }
