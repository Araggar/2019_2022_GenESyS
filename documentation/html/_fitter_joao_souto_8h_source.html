61 double FitterJoaoSouto::square_error(F f, Args ... args)
62 {
63  if (!_stats)
64  return std::numeric_limits<double>::max();
65 
66  double error = 0;
67  double range = (_stats->max() - _stats->min()) / _stats->histogramNumClasses();
68  double i = _stats->min();
69  double j = i + range;
70  double total_elements = _stats->numElements();
71 
72  Traits<Integrator_if>::Implementation integrator;
73 
74  for (auto c = 0; c < _stats->histogramNumClasses(); c++, i = j, j += range)
75  {
76  double area = integrator.integrate(i, j, f, args...);
77  error += pow(_stats->histogramClassFrequency(c) / total_elements - area, 2);
78  }
79 
80  return error;
81 }
